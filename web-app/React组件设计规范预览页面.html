💻 代码示例

1. 标准页面模板

import React, { useState } from 'react'; import { PageLayout, Card, Button, Input } from '@/components/ui'; const NewFeaturePage = () => { const [loading, setLoading] = useState(false); const handleAction = async () => { setLoading(true); try { // 业务逻辑 } finally { setLoading(false); } }; return ( <PageLayout title="功能页面" showBack={true}> <PageLayout.Content> <Card title="功能区域" className="mb-4"> <Button variant="primary" loading={loading} onClick={handleAction} > 执行操作 </Button> </Card> </PageLayout.Content> </PageLayout> ); }; export default NewFeaturePage;

2. 表单页面模板

import React, { useState } from 'react'; import { PageLayout, Card, Input, Select, Button } from '@/components/ui'; const FormPage = () => { const [formData, setFormData] = useState({ name: '', category: '', description: '' }); return ( <PageLayout title="数据录入" showBack={true}> <PageLayout.Content> <Card title="基本信息" className="mb-4"> <div className="space-y-4"> <Input label="产品名称" value={formData.name} onChange={(e) => setFormData({...formData, name: e.target.value})} placeholder="请输入产品名称" /> <Select label="产品类别" value={formData.category} onChange={(value) => setFormData({...formData, category: value})} options={[ { value: 'fruit', label: '水果' }, { value: 'vegetable', label: '蔬菜' } ]} /> </div> </Card> <div className="flex space-x-2"> <Button variant="secondary" className="flex-1">取消</Button> <Button variant="primary" className="flex-1">保存</Button> </div> </PageLayout.Content> </PageLayout> ); };

📚 组件API文档

Button 组件

属性 类型 默认值 说明
variant string 'primary' 按钮类型：primary | secondary | success | warning | danger
size string 'md' 按钮大小：sm | md | lg
loading boolean false 是否显示加载状态
disabled boolean false 是否禁用

Card 组件

属性 类型 默认值 说明
title string - 卡片标题
className string '' 自定义样式类
children ReactNode - 卡片内容
