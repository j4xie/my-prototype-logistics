package com.cretas.rules.validation

import com.cretas.aims.entity.MaterialBatch
import java.time.LocalDate

rule "原材料批次验证"
    salience 10
    when
        $batch : MaterialBatch(
            batchNumber == null ||
            batchNumber.length() < 5,
            quantity <= 0,
            expirationDate < LocalDate.now()
        )
    then
        $batch.setValid(false);
        $batch.addError("批次验证失败");
        // 触发告警通知
        alertService.send($batch);
end
条件说明
批次号格式检查
批次号不能为空且长度必须大于等于5个字符
数量有效性
批次数量必须大于0
过期日期检查
批次过期日期必须晚于当前日期
执行动作
设置批次为无效状态
