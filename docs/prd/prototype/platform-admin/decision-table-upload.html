rule "温度等级_规则1_优秀"
    salience 100
    when
        $batch : MaterialBatch(
            temperature < -25,
            productType == "冷冻海鲜"
        )
    then
        $batch.setGrade("优秀");
        $batch.setProcessing("正常入库");
end

rule "温度等级_规则2_合格"
    salience 99
    when
        $batch : MaterialBatch(
            temperature >= -25 && temperature <= -18,
            productType == "冷冻海鲜"
        )
    then
        $batch.setGrade("合格");
        $batch.setProcessing("正常入库");
end
